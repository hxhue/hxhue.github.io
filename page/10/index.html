CUDA дёӯ SFINAE зҡ„еқ‘

еҸ‘иЎЁдәҺпјҡ 2024-06-22 жӣҙж–°дәҺпјҡ 2024-06-28

TL;DR

еҰӮжһңжғіиҰҒеҶҷдёҖдәӣ traitsпјҢиҖҢдё”ж¶үеҸҠеҲ°дәҶеҮҪж•°е’ҢеҸҳйҮҸпјҲиҖҢдёҚжҳҜзұ»еһӢпјүпјҢеҲҷйңҖиҰҒеҗҢж—¶еҠ дёҠ __device__ е’Ң __host__ д»ҘзЎ®дҝқе®ғеңЁ device дҫ§е’Ң host дҫ§йғҪеҸҜи§ҒгҖӮеҗҰеҲҷдјҡеӣ дёә SFINAE дёҚжҠҘй”ҷпјҲе®һйҷ…дёҠеҸҜиғҪжҳҜеңЁ device дҫ§жүҫдёҚеҲ°еҮҪж•°жҲ–еҸҳйҮҸпјүпјҢиҖҢдё”дёҚзҹҘйҒ“дёәд»Җд№Ҳз»“жһңдёҚеҜ№гҖӮ

жҸҸиҝ°

йЎ№зӣ®з”Ёзҡ„иҜӯиЁҖж ҮеҮҶжҳҜ C++17пјҢatomicAdd жҳҜ CUDA дёӯзҡ„дёҖж—ҸеҮҪж•°пјҢиғҪеҜ№ж”ҜжҢҒзұ»еһӢиҝӣиЎҢеҺҹеӯҗж“ҚдҪңпјҢе…¶еҸӮж•°дёәдёҖдёӘең°еқҖе’ҢдёҖдёӘеҖјпјҢжҲ‘жғіиҰҒеҲӨж–ӯ atomicAdd жҳҜеҗҰжңүжҹҗдёӘеҹәжң¬ж•°жҚ®зұ»еһӢеҜ№еә”зҡ„йҮҚиҪҪзүҲжң¬гҖӮ

template <typename T, typename U = std::decay_t<T>>
__device__ __host__ auto atomicAddTest(T&&)
    -> std::void_t<decltype(atomicAdd((U*)(nullptr), std::declval<U>()))>;

__device__ __host__ auto atomicAddTest(...) -> int;

template <typename T>
__device__ __host__ static inline constexpr auto atomicAddable =
    std::is_same_v<decltype(atomicAddTest(std::declval<T>())), void>;

static_assert(atomicAddable<float>);
static_assert(atomicAddable<unsigned>);

зңӢиө·жқҘеҫҲжӯЈеёёеҜ№дёҚеҜ№пјҹдҪҶжҳҜеҰӮжһңжғіиҰҒеҗҢж—¶еңЁ host дҫ§е’Ң device дҫ§жқҘдҪҝз”Ё atomicAddable еҸҳйҮҸпјҢдҪҶжҳҜеҸҲеҝҳи®°з»ҷиҝҷдәӣеҮҪж•°еҠ  __device__ __host__пјҢе°ұдјҡжңү calling device function from host зҡ„й”ҷиҜҜпјҢз„¶еҗҺеӣ дёә SFINAE дёҚжҠҘй”ҷпјҢдёӢйқўзҡ„зЁӢеәҸе®һйҷ…дҪҝз”Ёж—¶жҖ»жҳҜиө°дәҶдёҚдёҖж ·зҡ„еҲҶж”ҜпјҢеҚҠеӨ©дёҚзҹҘйҒ“еҺҹеӣ еңЁе“ӘйҮҢпјҒеҗҺжқҘеҠ дәҶ static_assertпјҢеҗҢж—¶жҡӮж—¶жҖ§ең°е°Ҷ SFINAE дҝқжҠӨзҡ„йғЁеҲҶж”ҫеҲ°еҮҪж•°дҪ“йҮҢйқўеҺ»пјҢжүҚйҖҡиҝҮ LSP дәҶи§ЈеҲ°е…·дҪ“зҡ„еҮәй”ҷеҺҹеӣ гҖӮ
