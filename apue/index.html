void

{

{
    struct stat buf;
    const char *ptr;
    for (int i = 1; i < argc; ++i)
        {

            if (lstat (argv[i], &buf) < 0)

    if (val & O_APPEND)
        printf (", append");
    if (val & O_NONBLOCK)
        printf (", nonblocking");
    if (val & O_SYNC)
        printf (", synchronous writes");
#if !defined(_POSIX_C_SOURCE) && defined(O_FSYNC) && (O_FSYNC != O_SYNC)
    if (val & O_FSYNC)
        printf (", synchronous writes");
#endif
    putchar ('\n');
    exit (0);
}
