parse_git_branch() {
    if ! [[ $(git rev-parse --is-bare-repository 2>/dev/null) == true ]]; then
        local BRANCH="$(git branch 2>/dev/null | sed -e '/^[^*]/d' -e 's/* \(.*\)/\1/')"
        [[ "${BRANCH}" == "" ]] && return 0
        local MSG="$(git status 2> /dev/null | tail -n1)"
        if [[ "$MSG" != "nothing to commit"* ]]; then
            echo -n "(${BRANCH}*) "
        else
            echo -n "(${BRANCH}) "
        fi
    fi
}
