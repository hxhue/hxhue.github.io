#!/bin/bash
is_inside_vscode() {
  if ([[ "$TERM" == "" ]] || [[ "$TERM" == "dumb" ]]) && [[ "$PATH" == *".vscode-server/"* ]]; then
    return 0
  fi
  return 1
}

print_header() {
  echo "############################################################"
  echo "#   This is the temp file for showing pre-commit output.   #"
  echo "############################################################"
  echo
}

entry_point() { :; }

if is_inside_vscode; then
  tmp=$(mktemp)
  code $tmp
  print_header >$tmp
  if ! entry_point >>$tmp 2>&1; then
    echo "pre-commit failed, please check $tmp" && false
  fi
else
  entry_point
fi
